# SINGLE PLAYER DONE # MULTI PLAYER TO BE COMPLETED import pygame import sys import random from pygame.locals import * pygame.init() clock = pygame.time.Clock() font = pygame.font.SysFont(None, 55) white = [255,255,255] black = [0,0,0] green = [0,255,0] def choose_figure(): num = random.randrange(1,4) if num == 1: num = "ROCK" elif num == 2: num = "PAPER" else: num = "SCISSORS" return num def welcome(): DISPLAYSURF.fill(white) text_screen("WELCOME TO ROCK PAPER SCISSORS",black,80,50) text_screen("You can play :" , black , 330,150 ) # SINGLE PLAYER pygame.draw.rect(DISPLAYSURF,green,(270,270 , 350,90)) text_screen("SINGLE PLAYER",black,300,300) # MULTIPLAYER pygame.draw.rect(DISPLAYSURF,green,(270,420 , 350,90)) text_screen("MULTI-PLAYER",black,300,450) update() while True: for event in pygame.event.get(): if event.type == QUIT: pygame.quit() sys.exit() if event.type == MOUSEBUTTONDOWN: pos = pygame.mouse.get_pos() x = pos[0] y = pos[1] if x >= 270 and x <= 620 and y >= 270 and y <= 360: gameloop() def recreate(): DISPLAYSURF.fill(white) # BACK BUTTON text_screen("BACK",black,10,10) # ROCK BUTTON pygame.draw.rect(DISPLAYSURF,green,(680,470 , 170,90)) text_screen("ROCK",black,710,500) # PAPER BUTTON pygame.draw.rect(DISPLAYSURF,green,(480,470 , 185,90)) text_screen("PAPER",black,510,500) #
