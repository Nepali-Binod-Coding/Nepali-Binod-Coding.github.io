Rock paper scissors code walkthrough in pygame.

In this blog I will walk you through the whole code of rock paper scissors game in pygame (A python library used to make games in python). Rock paper scissors is a famous game. Mainly played amongst kids. Rock paper scissors is a hand game usually played between two people, in which each player simultaneously forms one of three shapes with an outstretched hand. These shapes are "rock" (a closed fist), "paper" (a flat hand), and "scissors" (a fist with the index finger and middle finger extended, forming a V).

Importing libraries & Initializing

import pygame
import sys
import random
from pygame.locals import *

pygame.init()

Creating global variables

font = pygame.font.SysFont(None, 55)
white = (255,255,255)
black = (0,0,0)
green = (0,255,0)
DISPLAYSURF = pygame.display.set_mode((900,600))
pygame.display.set_caption("ROCK PAPER SCISSORS")

Creating basic functions for the program

def text_screen(text,color,x,y):
screen_text = font.render(text, True , color) DISPLAYSURF.blit(screen_text, [x,y]) update()

pygame.display.update()

Creating interface of the game

def recreate(): DISPLAYSURF.fill(white) text_screen("BACK",black,10,10) pygame.draw.rect(DISPLAYSURF,green,(680,470 , 170,90)) text_screen("ROCK",black,710,500) pygame.draw.rect(DISPLAYSURF,green,(480,470 , 185,90)) text_screen("PAPER",black,510,500) pygame.draw.rect(DISPLAYSURF,green,(230,470 , 235,90)) text_screen("SCISSOR",black,260,500) update()
def sorry(): DISPLAYSURF.fill(white) text_screen("Sorry we are working on this mode",black,130,10) pygame.draw.rect(DISPLAYSURF,green,(350,300,140,90)) text_screen("OK!!",black,380,330)

Creating the main game backend

def gameloop(): recreate() while True: for event in pygame.event.get(): if event.type == QUIT: pygame.quit() sys.exit() if event.type == MOUSEBUTTONDOWN: pos = pygame.mouse.get_pos() x = pos[0] y = pos[1] if x >= 230 and x <= 470 and y >= 470 and y <= 550 : recreate() text_screen("SCISSORS" , black , 590,250) num = choose_figure() text_screen(num , black , 100,250) if num == "ROCK": text_screen("LOST",black , 400,250) elif num == "PAPER": text_screen("WON" , black , 400,250) else: text_screen("TIE" , black , 400,250) elif x >= 480 and x <= 670 and y >= 470 and y <= 550 : recreate() text_screen("PAPER" , black , 590,250) num = choose_figure() text_screen(num , black , 100,250) if num == "ROCK": text_screen("WON",black , 400,250) elif num == "PAPER": text_screen("TIE" , black , 400,250) else: text_screen("LOST" , black , 400,250) elif x >= 680 and x <= 870 and y >= 470 and y <= 550 : recreate() text_screen("ROCK" , black , 590,250) num = choose_figure() text_screen(num , black , 100,250) if num == "ROCK": text_screen("TIE",black , 400,250) elif num == "PAPER": text_screen("LOST" , black , 400,250) else: text_screen("WON" , black , 400,250) elif x>=0 and x <= 120 and y >= 0 and y <= 45: welcome()

Last function to complete the game.

def choose_figure(): num = random.randrange(1,4) if num == 1: num = "ROCK" elif num == 2: num = "PAPER" else: num = "SCISSORS" return num
